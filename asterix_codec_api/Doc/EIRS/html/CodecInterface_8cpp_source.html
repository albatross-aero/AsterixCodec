379 void cleanDirectory(const char* directoryPath) {
380 
381  DIR *directory; // Representing the directory stream
382  struct dirent *entry; // Represent next directory entry in the directory stream, pointed by directory
383  char filePathToDelete[PATH_MAX];
384 
385  //Open directory and read all entry, if a entry is different from '.' and '..', it is a file to remove
386  directory = opendir(directoryPath);
387  if (directory == NULL) {
388  return;
389  }
390 
391  while ((entry = readdir(directory)) != NULL) {
392 
393  if (strcmp(entry->d_name, ".") && strcmp(entry->d_name, "..")) {
394  snprintf(filePathToDelete, (size_t) PATH_MAX, "%s/%s", directoryPath, entry->d_name); // create path of file to remove
395  remove(filePathToDelete);
396  }
397  }
398 
399  closedir(directory);
400 }
